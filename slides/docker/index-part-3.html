    return new Promise(function (yes, no) {
      try {
        mongo.dbo.collection('scans').findOne({}, function (err, res) {
          if (err) throw err
          yes(res)
        })
      } catch (err) {
        console.log('ERROR mongo SET', err)
        no(err)
      }
    })
  }
